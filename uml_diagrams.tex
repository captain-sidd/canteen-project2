\subsection{Integration Instructions}
\begin{itemize}[nosep]
  \item Add \texttt{\usepackage{tikz}} and \texttt{\usetikzlibrary{positioning}} to the LaTeX preamble.
  \item Insert diagrams into \texttt{SRS.tex} with \texttt{\input{uml_diagrams.tex}} at the desired location.
  \item If compiling from a root document, ensure \texttt{uml_diagrams.tex} is in the same directory or update the path accordingly.
  \item For PlantUML, install PlantUML and render \texttt{uml_diagrams.puml} to SVG/PNG using:\newline
    \texttt{plantuml -tsvg uml_diagrams.puml}
  \item Include rendered PlantUML images in LaTeX with \texttt{\includegraphics[width=\linewidth]{uml_diagrams.svg}}.
  \item For Mermaid, install \texttt{mmdc} and render \texttt{uml_diagrams.mmd} using:\newline
    \texttt{mmdc -i uml_diagrams.mmd -o uml_diagrams.svg}
  \item Use \texttt{\usepackage{graphicx}} in LaTeX and include Mermaid output with \texttt{\includegraphics[width=\linewidth]{uml_diagrams.svg}}.
\end{itemize}
